#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int N;
vector<int> A;

void solve() {
	cin >> N;
	A.resize(N);
	for (int n = 0; n < N; n++) {
		cin >> A[n];
		A[n]--;
	}
	int count = 0;
	for (int n = 0; n < N; n++) {
		if (A[A[n]] == n)count++;
	}
	cout << count / 2 << endl;
	return;
}
