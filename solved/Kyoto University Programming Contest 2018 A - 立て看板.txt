using namespace std;

int N;
vector<int> S, A;

int main() {
	cin >> N;
	S.resize(N);
	A.resize(N);
	for (int n = 0; n < N; n++)cin >> S[n];
	for (int n = 0; n < N; n++)cin >> A[n];
	int ans = 0;
	for (int n = 0; n < N; n++)ans = max(ans, S[n] * A[n]);
	cout << ans << endl;
	return 0;
}
