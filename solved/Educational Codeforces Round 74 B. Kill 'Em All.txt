using namespace std;
 
int main() {
	int Query;
	cin >> Query;
	for (int q = 0; q < Query; q++) {
		int N, R;
		cin >> N >> R;
		vector<int> A;
		A.resize(N);
		for (int n = 0; n < N; n++)cin >> A[n];
		deque<int> D;
		sort(A.begin(), A.end(), [](int a, int b) {
			return a < b;
		});
		for (int n = 0; n < N; n++)D.push_back(A[n]);
		int ans = 0;
		while (!D.empty()) {
			ans++;
			int rightest = D.back();
			while (!D.empty()) {
				if (D.back() == rightest)D.pop_back();
				else break;
			}
			while (!D.empty()) {
				if (D.front() <= R * ans)D.pop_front();
				else break;
			}
		}
		cout << ans << endl;
	}
	return 0;
}
