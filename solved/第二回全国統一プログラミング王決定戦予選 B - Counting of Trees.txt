ll N;

int main() {
	cin >> N;
	D.resize(N);
	for (ll n = 0; n < N; n++) {
		cin >> D[n];
	}
	check.resize(1e5 + 1,0);
	for (ll n = 0; n < N; n++) {
		check[D[n]]++;

		if (D[n] == 0)ans *= 0;
		ans *= check[D[n]-1];
		ans %= 998244353;
	}
	cout << ans << endl;
	return 0;
}
