long long MOD = 1e9 + 7;

int N;
vector<int> res;

int main() {
	cin >> N;
	while (N != 0) {
		if (N & 1) {
			res.push_back(1);
			N--;
			N /= -2;
		}
		else {
			res.push_back(0);
			N /= -2;
		}
	}
	if (res.size() == 0)cout << 0 << endl;
	else {
		for (int n = res.size() - 1; n >= 0; n--)cout << res[n];
		cout << endl;
	}
	return 0;
}
