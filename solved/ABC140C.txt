int N;

void solve() {
	long long ans = 0;
	cin >> N;
	vector<int> B;
	B.resize(N);
	for (int n = 0; n < N-1; n++) {
		cin >> B[n];
	}
	ans += B[0];
	for (int n = 1; n < N-1; n++) {
		ans += min(B[n-1],B[n]);
	}
	ans += B[N -2];
	cout << ans << endl;
	return;
}

		solve();
		cout << "####################" << endl;
	}
#else
	solve();
#endif
	return 0;
}
