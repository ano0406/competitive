	solve();
	return 0;

using namespace std;

int N, K;

void solve() {
	cin >> N >> K;
	int ans = -1;
	int m = 0;
	for (int n = 1; n <= N; n++) {
		int a;
		cin >> a;
		if (a < K) {
			if (m < a) {
				ans = n;
				m = a;
			}
		}
	}
	cout << ans << endl;
	return;
}
