long long MOD = 1e9 + 7;

vector<int> A;

int main() {
	A.resize(5);
	for (int i = 0; i < 5; i++)cin >> A[i];
	set<int> res;
	for (int code = 0; code < 32; code++) {
		int temp = 0;
		int c = 0;
		for (int n = 0; n < 5; n++) {
			if (code & (1 << n)) {
				temp += A[n];
				c++;
			}
		}
		if(c == 3)res.insert(temp);
	}
	vector<int> ans;
	for (int num : res) {
		ans.push_back(num);
	}
	cout << ans[ans.size()-3] << endl;
	return 0;
}
