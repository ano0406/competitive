int N;
typedef pair<int, int> pii;
vector <pii> v1;
vector <pii> v2;

void solve() {
	cin >> N;
	for (int i = 0; i < N; i++) {
		int a, b;
		cin >> a >> b;
		v1.push_back(pii(a, b));
	}
	sort(v1.begin(), v1.end());
	for (int i = 0; i < N; i++) v2.push_back(pii(v1[i].first+v1[i].second,v1[i].first-v1[i].second));
	sort(v2.begin(),v2.end());
	int ans = 1;
	int end = v2[0].first;
	for(int i = 1;i < N;i++){
		if (v2[i].second >= end) {
			ans++;
			end = v2[i].first;
		}
	}
	cout << ans;
}
