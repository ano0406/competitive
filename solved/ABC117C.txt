const long long MOD = 1e9 + 7;
typedef long long ll;
typedef long double ld;
typedef pair<ll, ll> pll;
typedef pair<ld, ll> pdl;
typedef pair<ld, ld> pdd;
//typedef boost::multiprecision::cpp_int bigint;

ll N,M;
vector<ll> A, dif;

int main() {
	cin >> M >> N;
	A.resize(N);
	for (ll n = 0; n < N; n++)cin >> A[n];
	sort(A.begin(), A.end(), [](ll a, ll b) {return a < b; });
	dif.resize(N - 1);
	for (ll n = 1; n < N; n++)dif[n - 1] = A[n] - A[n - 1];
	sort(dif.begin(), dif.end(), [](ll a, ll b) {return a < b; });
	ll ans = 0;
	for (ll n = 0; n < N - M; n++)ans += dif[n];
	cout << ans << endl;
	return 0;
}
