int main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	ll N, M;
	cin >> N >> M;
	VLL A(N);
	for (ll n = 0; n < N; n++)cin >> A[n];
	for (ll k = 1; k <= M; k++) {
		for (ll i = 0; i < N-1; i++) {
			if (A[i] % k > A[i + 1] % k)swap(A[i], A[i + 1]);
		}
	}
	for (ll n = 0; n < N; n++) {
		cout << A[n] << "\n";
	}
	return 0;
}
