template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

int N;

int main() {
	cin >> N;
	int s = N / 10;
	int single = N - s * 10;
	cout << min(s * 100 + single * 15, (s + 1) * 100) << endl;
	return 0;
}
