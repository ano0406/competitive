void solve() {
	int N;
	cin >> N;
	vector<int> X;
	X.resize(N);
	for (int n = 0; n < N; n++)cin >> X[n];
	sort(X.begin(), X.end());
	set<int> prime;
	prime.insert(2);
	prime.insert(3);
	for (int p = 4; p <= X[N - 1]; p++) {
		bool flag = false;
		for (int n = 2; n * n <= p; n++) {
			if (p % n == 0) {
				flag = true;
				break;
			}
		}
		if (!flag)prime.insert(p);
	}
	for (int n = 0; n < N; n++) {
		int oldx = X[n];
		X[n] = 1;
		while (true) {
			if (prime.find(oldx - 2) == prime.end())break;
			oldx -= 2;
			X[n]++;
		}
	}
	int g = 0;
	for (int n = 0; n < N; n++) {
		g ^= X[n];
	}
	if (g != 0)cout << "An" << endl;
	else cout << "Ai" << endl;
	return;
}

		solve();
		cout << "####################" << endl;
	}
#else
	solve();
#endif
	return 0;
}
