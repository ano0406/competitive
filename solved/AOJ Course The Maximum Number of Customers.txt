int main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	ll N, T;
	cin >> N >> T;
	VLL imos(T+1, 0);
	for (ll n = 0; n < N; n++) {
		ll s, t;
		cin >> s >> t;
		imos[s]++;
		imos[t]--;
	}
	ll ans = imos[0];
	for (ll t = 1; t < T; t++) {
		imos[t] += imos[t - 1];
		ans = max(ans, imos[t]);
	}
	cout << ans << "\n";
	return 0;
}
