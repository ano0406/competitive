using namespace std;
long long MOD = 1e9 + 7;

string S, T;
vector<int> scount, tcount;

int main() {
	cin >> S >> T;
	scount.resize(26, 0);
	tcount.resize(26, 0);
	for (int n = 0; n < S.size(); n++)scount[S[n] - 'a']++;
	for (int n = 0; n < T.size(); n++)tcount[T[n] - 'a']++;
	sort(scount.begin(), scount.end(), [](int a, int b) {
		return a < b;
	});
	sort(tcount.begin(), tcount.end(), [](int a, int b) {
		return a < b;
	});
	bool flag = true;
	for (int c = 0; c < 26; c++) {
		if (scount[c] != tcount[c]) {
			flag = false;
			break;
		}
	}
	if (flag)cout << "Yes" << endl;
	else cout << "No" << endl;
	return 0;
}
