using namespace std;

typedef pair<int, int> pii;
vector<pii> A;
vector<int> ans;
int N;

int main() {
	cin >> N;
	A.resize(N);
	ans.resize(N);
	for (int n = 0; n < N; n++) {
		cin >> A[n].second;
		A[n].first = n;
	}
	sort(A.begin(), A.end(), [](pii a, pii b) {
		return a.second < b.second;
	});
	int num = 0;
	ans[A[0].first] = num;
	for (int n = 1; n < N; n++) {
		if (A[n].second == A[n - 1].second)ans[A[n].first] = num;
		else {
			num++;
			ans[A[n].first] = num;
		}
	}
	for (int n = 0; n < N; n++)cout << ans[n] << endl;
	return 0;
}
