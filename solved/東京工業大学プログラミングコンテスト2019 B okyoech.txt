void solve() {
	int Query;
	cin >> Query;
	for (int query = 0; query < Query; query++) {
		string S;
		cin >> S;
		//okyo
		int n = 0;
		for (; n < (int)S.size() - 3; n++) {
			if (S.substr(n,4) == string("okyo"))break;
		}
		if (n >= (int)S.size() - 3) {
			cout << "No" << endl;
			continue;
		}
		int n2 = n + 4;
		for (; n2 < (int)S.size() - 2; n2++) {
			if (S.substr(n2,3) == string("ech"))break;
		}
		if (n2 >= (int)S.size() - 2) {
			cout << "No" << endl;
			continue;
		}
		cout << "Yes" << endl;
		continue;
	}
}
