void solve() {
	int A, B, T;
	cin >> A >> B >> T;
	if ((T - B) % (B - A) == 0) {
		cout << T << endl;
		return;
	}
	int tmp = (T - B) / (B - A);
	cout << B + (tmp + 1) * (B - A) << endl;
	return;
}

		solve();

	}
#else
	solve();
#endif
	return 0;
}
