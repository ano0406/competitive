template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

int N, S, T;

int main() {
	cin >> N >> S >> T;
	if (S > T)cout << -1 << endl;
	else if (S == T)cout << 0 << endl;
	else {
		int c = 0;
		while (T > S) {
			T /= 2;
			c++;
		}
		if (S == T)cout << c << endl;
		else cout << -1 << endl;
	}
	return 0;
}
