#define DEBUGTIME 1

#include <iostream>
#include <vector>
#include <limits.h>
#include <algorithm>
#include <string>
#include <math.h>
#include <limits.h>
#include <queue>
#include <map>
#include <set>
#include <iomanip>
#include <bitset>
#include <cassert>
#include <random>
using namespace std;

int W, H;
vector<int> C;

void solve() {
	cin >> H >> W;
	C.resize(W,0);
	for (int h = 0; h < H; h++) {
		for (int x = 0; x < W;x++) {
			int t;
			cin >> t;
			if (t == 1)C[x] |= (1 << h);
		}
	}
	int ans = 0;
	for (int r = 0; r < (1 << H); r++) {
		int tempans = 0;
		for (int w = 0; w < W; w++) {
			int cc = C[w] ^ r;
			int counter = 0;
			for (int h = 0; h < H; h++) {
				counter += (cc >> h) % 2;
			}
			tempans += max(counter,H-counter);
		}
		ans = max(ans, tempans);
	}
	cout << ans << endl;
	return;
}

//////////////////////////////
//////////////////////////////

int main() {
#ifdef DEBUGTIME
	for (int time = 0; time < DEBUGTIME; time++) {
		solve();
		cout << "####################" << endl;
	}
#else
	solve();
#endif
	return 0;
}
