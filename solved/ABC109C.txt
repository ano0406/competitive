template<class T>
T GCD(T a, T b) {
	if (b == 0)return a;
	return GCD(b, (T)(a % b));
}

ll N, X;
vector<ll> towns;

int main() {
	cin >> N >> X;
	towns.resize(N + 1);
	towns[0] = X;
	for (ll n = 1; n <= N; n++)cin >> towns[n];
	sort(towns.begin(), towns.end(), [](ll a, ll b) {
		return a < b;
	});
	ll d = towns[1] - towns[0];
	for (ll n = 1; n < N; n++) {
		d = GCD(d, towns[n + 1] - towns[n]);
	}
	cout << d << endl;
	return 0;
}
