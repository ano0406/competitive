#include <iostream>
#include <math.h>
#include <algorithm>
#include <vector>
using namespace std;

void solve();

int main() {
	solve();
	return 0;
}

//////////////////////////////////////////////////
//////////////////////////////////////////////////

int N;
vector<int> arr;

void solve() {
	cin >> N;
	for (int n = 0; n < N; n++) {
		int next;
		cin >> next;
		sort(arr.begin(),arr.end());
		bool flag = false;
		for (auto itr = arr.begin(); itr != arr.end(); itr++) {
			if (*itr >= next) {
				*itr = next;
				flag = true;
				break;
			}
		}
		if (!flag)arr.push_back(next);
	}
	cout << arr.size() << endl;
	return;
}
