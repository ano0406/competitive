#include <iostream>
#include <math.h>
using namespace std;

int main(){
  int M,K;
  cin >> M >> K;
  if(K >= pow(2,M)){
    cout << -1 << endl;
    return 0;
  }
  if(M == 0){
    cout << "0 0" << endl;
    return 0;
  }
  if(M == 1){
    if(K == 0)cout << "0 0 1 1" << endl;
    else cout << -1 << endl;
    return 0;
  }
  if(K == 0){
    for(int i = 0;i < pow(2,M);i++)cout << i << " ";
    for(int i = pow(2,M)-1;i >= 1;i--)cout << i << " ";
    cout << 0 << endl;
    return 0;
  }
  for(int i = 0;i < pow(2,M);i++){
    if(i == K)continue;
    cout << i << " ";
  }
  cout << K << " ";
  for(int i = pow(2,M)-1;i >= 0;i--){
    if(i == K)continue;
    cout << i << " ";
  }
  cout << K << endl;
  return 0;
}
