#include <iostream>
#include <math.h>
#include <string>

using namespace std;

void solve();

int main() {
	solve();
	return 0;
}

//////////////////////////////////////////////////
//////////////////////////////////////////////////

int D, G;
int p[10], c[10];
void solve() {
	cin >> D >> G;
	for (int i = 0; i < D; i++)cin >> p[i] >> c[i];
	int max = 1000;
	for (int b = 0; b < pow(2, D); b++) {   //1のところだけコンプボーナスをとる
		int dif = G,num = 0;
		for (int i = 0; i < D; i++) {
			int t = b & (int)pow(2, i);
			if (t != 0) {
				dif -= (100 * (i + 1)*p[i] + c[i]);
				num += p[i];
			}
		}
		//ここから貪欲
		for (int i = D-1; i >= 0 && dif > 0 ; i--) {
			int t = b & (int)pow(2, i);
			if (t == 0) {
				int used = floor(dif/(100*(i+1)));
				if (used > p[i] - 1)used = p[i] - 1;
				dif -= 100 * (i + 1)*used;
				num += used;
			}
		}
		if (dif <= 0 && max > num)max = num;
	}
	cout << max;
	return;
}
