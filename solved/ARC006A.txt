template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

int E[6];
int B;
int L[6];
int check[6];   //0->一致 1->ボーナス数字 2->不一致	

int main() {
	for (int i = 0; i < 6; i++)cin >> E[i];
	cin >> B;
	for (int i = 0; i < 6; i++)cin >> L[i];
	for (int l = 0; l < 6; l++) {
		check[l] = 2;
		if (L[l] == B) {
			check[l] = 1;
			continue;
		}
		for (int e = 0; e < 6; e++) {
			if (L[l] == E[e])check[l] = 0;
		}
	}
	int count0 = 0;
	int count1 = 0;
	for (int l = 0; l < 6; l++) {
		if (check[l] == 0)count0++;
		else if (check[l] == 1)count1++;
	}
	if (count0 == 6)cout << 1 << endl;
	else if (count0 == 5 && count1 == 1)cout << 2 << endl;
	else if (count0 == 5 && count1 == 0)cout << 3 << endl;
	else if (count0 == 4)cout << 4 << endl;
	else if (count0 == 3)cout << 5 << endl;
	else cout << 0 << endl;
	return 0;
}
