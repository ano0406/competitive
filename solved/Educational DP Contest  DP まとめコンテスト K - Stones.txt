template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

int N, K;
vector<int> A;
vector<int> G;

int main() {
	cin >> N >> K;
	A.resize(N);
	in(A);
	G.resize(K + 1,0);
	for (int k = 0; k <= K; k++) {
		set<int> s;
		for (int n = 0; n < N; n++) {
			if (k < A[n])continue;
			s.insert(G[k - A[n]]);
		}
		int g = 0;
		while (s.count(g) != 0)g++;
		G[k] = g;
	}
	if (G[K] == 0)cout << "Second" << endl;
	else cout << "First" << endl;
	return 0;
}
