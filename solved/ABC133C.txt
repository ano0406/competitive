int main() {
	solve();
	return 0;
}

#include <iostream>
#include <vector>
#include <limits.h>
#include <algorithm>
#include <iomanip>
#include <string>
#include <math.h>

using namespace std;

int L, R;
void solve() {
	cin >> L >> R;
	if (R - L >= 2018) {
		cout << 0 << endl;
		return;
	}
	L %= 2019;
	R %= 2019;
	if (R <= L) {
		cout << 0 << endl;
		return;
	}
	int ans = 2018;
	for (int i = L; i <= R; i++) {
		for (int j = i+1; j <= R; j++) {
			ans = min(ans, (i * j) % 2019);
		}
	}
	cout << ans << endl;
	return;
}
