#include <iostream>
#include <math.h>
#include <string>

using namespace std;

void solve();

int main() {
	solve();
	return 0;
}

//////////////////////////////////////////////////

string S;
void solve() {
	cin >> S;
	int nums[4];
	for (int i = 0; i < 4; i++)nums[i] = S[i] - '0';
	int a, b, c;
	for (a = 0; a <= 1; a++) {
		for (b = 0; b <= 1; b++) {
			for (c = 0; c <= 1; c++) {
				int temp = nums[0];
				temp = a == 0 ? temp + nums[1] : temp - nums[1];
				temp = b == 0 ? temp + nums[2] : temp - nums[2];
				temp = c == 0 ? temp + nums[3] : temp - nums[3];
				if (temp == 7) {
					cout << nums[0];
					if (a == 0)cout << "+";
					else cout << "-";
					cout << nums[1];
					if (b == 0)cout << "+";
					else cout << "-";
					cout << nums[2];
					if (c == 0)cout << "+";
					else cout << "-";
					cout << nums[3] << "=7";
					return;
				}
			}
		}
	}
}
