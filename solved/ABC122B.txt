using namespace std;

typedef pair<int, int> pii;
string S;

void solve() {
	cin >> S;
	int m = 0;
	for (int s = 0; s < S.size(); s++) {
		for (int e = s; e < S.size(); e++) {
			bool flag = true;
			for (int i = s; i <= e; i++) {
				if (S[i] != 'A' && S[i] != 'G' && S[i] != 'C' && S[i] != 'T') {
					flag = false;
					break;
				}
			}
			if (flag)m = max(m, e - s + 1);
		}
	}
	cout << m << endl;
	return;
}
