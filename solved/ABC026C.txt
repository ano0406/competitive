long long MOD = 1e9 + 7;

int main() {
	long long N;
	cin >> N;
	vector<long long> A;
	A.resize(N);
	for (int n = 0; n < N; n++)cin >> A[n];
	sort(A.begin(), A.end(), [](long long a, long long b) {
		return a > b;
	});
	long long ans = 0;
	for (int n = 0; n < N; n++) {
		if (n & 1)ans -= A[n] * A[n];
		else ans += A[n] * A[n];
	}
	cout << setprecision(12)<< (long double)ans * acos(-1) << endl;
	return 0;
}
