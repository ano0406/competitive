int main() {
	int Query;
	cin >> Query;
	for (int q = 0; q < Query; q++) {
		int a, b, c;
		cin >> a >> b >> c;
		int k2 = min(b / 1, c / 2);
		int ans = k2 * 3;
		b -= k2;
		c -= k2 * 2;
		int k1 = min(a, b / 2);
		ans += k1 * 3;
		a -= k1;
		b -= 2 * k1;
		cout << ans << endl;
	}
	return 0;
}
