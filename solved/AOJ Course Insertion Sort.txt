ll N;
vector<ll> A;

int main() {
	cin >> N;
	A.resize(N);
	for (ll n = 0; n < N; n++)cin >> A[n];
	for (ll n = 0; n < N - 1; n++)cout << A[n] << " ";
	cout << A[N - 1] << endl;
	for (ll i = 1; i < N; i++) {
		ll v = A[i];
		ll j = i - 1;
		while (j >= 0 && A[j] > v) {
			A[j + 1] = A[j];
			j--;
		}
		A[j + 1] = v;
		for (ll n = 0; n < N - 1; n++)cout << A[n] << " ";
		cout << A[N - 1] << endl;
	}
	return 0;
}
