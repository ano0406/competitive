template<class T>
void input(T& I) {
	cin >> I;
	return;
}

template<class T>
void input(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void output(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

void output(string S) {
	cout << S << endl;
	return;
}

template<class T>
void output(T& O) {
	cout << O << endl;
	return;
}

int main() {
	string S;
	input(S);
	if (S == string("Sunny"))output(string("Cloudy"));
	else if (S == string("Cloudy"))output(string("Rainy"));
	else output(string("Sunny"));
	return 0;
}
