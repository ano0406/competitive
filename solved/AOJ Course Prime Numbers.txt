bool isPrime(int n) {
	for (int i = 2; i <= floor(sqrt(n)); i++) {
		if (n % i == 0)return false;
	}
	return true;
}

int main() {
	ll N;
	cin >> N;
	ll ans = 0;
	for (ll n = 0; n < N; n++) {
		ll a;
		cin >> a;
		if (isPrime(a))ans++;
	}
	cout << ans << endl;
	return 0;
