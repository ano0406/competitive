template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

int N, K;
vector<int> A;

int main() {
	cin >> N >> K;
	A.resize(N);
	in(A);
	sort(A.begin(), A.end(), [](int a, int b) {return a > b; });
	int temp = 0;
	int ans = 0;
	while (temp < K && ans < N) {
		temp += A[ans];
		ans++;
	}
	if (ans == N && temp < K)cout << -1 << endl;
	else cout << ans << endl;
	return 0;
}
