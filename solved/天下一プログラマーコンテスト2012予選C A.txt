//素数判定
//O(√n)
bool isPrime(int n) {
	for (int i = 2; i <= floor(sqrt(n)); i++) {
		if (n%i == 0)return false;
	}
	return true;
}

//n未満の自然数のでのエラトステネスの篩
//arr[n]=true -> 素数
//O(nlog(logn)) , 実質O(n)
bool* eratosthenes(int n) {
	bool* array = new bool[n];
	for (int i = 0; i < n; i++)array[i] = true;
	for (int i = 2; i < n; i++) {
		if (!array[i])continue;
		int t = 2*i;
		while (t < n) {
			array[t] = false;
			t += i;
		}
	}
	return array;
}

void solve() {
	int N;
	cin >> N;
	bool* arr = eratosthenes(N);
	int ans = 0;
	for (int n = 2; n < N; n++) {
		if (arr[n])ans++;
	}
	delete arr;
	cout << ans << endl;
	return;
}
