template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

int N;

int main() {
	cin >> N;
	int ans = 0;
	for (int n = 0; n < N; n++) {
		int a, b;
		cin >> a >> b;
		ans += a * b;
	}
	cout << (int)(ans * 1.05) << endl;
	return 0;
}
