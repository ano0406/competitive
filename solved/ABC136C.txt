int N;
vector<int> H;

int main() {
	cin >> N;
	H.resize(N);
	for (int n = 0; n < N; n++)cin >> H[n];
	H[0]--;
	for (int n = 1; n < N; n++) {
		if (H[n - 1] < H[n])H[n]--;
		else if (H[n - 1] == H[n])continue;
		else {
			cout << "No" << endl;
			return 0;
		}
	}
	cout << "Yes" << endl;
	return 0;
}
