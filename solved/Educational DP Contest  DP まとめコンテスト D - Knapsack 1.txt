template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

int N, Wmax;
vector<int> W, V;
vector<vector<long long>> DP;

int main() {
	cin >> N >> Wmax;
	W.resize(N + 1);
	V.resize(N + 1);
	for (int n = 1; n <= N; n++) {
		cin >> W[n] >> V[n];
	}
	DP.resize(N + 1, vector<long long>(Wmax+1,0));
	for (int n = 1; n <= N; n++) {
		for (int w = 0; w < W[n]; w++)DP[n][w] = DP[n - 1][w];
		for (int w = W[n]; w <= Wmax; w++) {
			DP[n][w] = max(DP[n - 1][w], DP[n - 1][w - W[n]] + V[n]);
		}
	}
	long long ans = 0;
	for (int w = 0; w <= Wmax; w++)ans = max(ans, DP[N][w]);
	cout << ans << endl;
	return 0;
}
