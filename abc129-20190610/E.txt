#include <iostream>
#include <vector>
#include <string>
using namespace std;

vector<long long> dp[2];   //dp[0]:–¢Šm’è dp[1]:Šm’è
vector<long long> L;
string S;
int MOD = 1e9 + 7;

void solve() {
	cin >> S;
	L.resize(S.size());
	for (int n = 0; n < S.size(); n++) L[n] = S[n] - '0';
	dp[0].resize(S.size());
	dp[1].resize(S.size());
	dp[0][0] = 2;
	dp[1][0] = 1;
	for (int n = 1; n < S.size(); n++) {
		dp[0][n] = (dp[0][n - 1] * (L[n] + 1)) % MOD;
		dp[1][n] = (3*dp[1][n-1]+L[n]*dp[0][n-1]) % MOD;
	}
	cout << (dp[0].back()+dp[1].back()) % MOD << endl;
	return;
}
