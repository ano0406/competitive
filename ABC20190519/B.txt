using namespace std;
using namespace boost::multiprecision;

string S;

void solve() {
	cin >> S;
	int a = stoi(S.substr(0, 2));
	int b = stoi(S.substr(2, 2));
	bool isMMYY = false;
	bool isYYMM = false;
	if (a >= 1 && a <= 12)isMMYY = true;
	if (b >= 1 && b <= 12)isYYMM = true;
	if (isYYMM && isMMYY)cout << "AMBIGUOUS" << endl;
	else if (isYYMM)cout << "YYMM" << endl;
	else if (isMMYY)cout << "MMYY" << endl;
	else cout << "NA" << endl;
	return;
}
