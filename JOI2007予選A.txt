int N;
void solve() {
	cin >> N;
	N = 1000 - N;
	int sum = 0;
	int coins[6] = {500,100,50,10,5,1};
	for (int i = 0; i < 6; i++) {
		sum += floor(N/coins[i]);
		N %= coins[i];
	}
	cout << sum << endl;
}
