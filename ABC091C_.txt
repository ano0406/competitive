int N;
typedef pair<int, int> ii;
vector<ii> A, B;
int smap[200][200];

void amap();

void solve() {
	cin >> N;
	for (int n = 0; n < N; n++) {
		int a, b;
		cin >> a >> b;
		A.push_back(ii(a,b));
	}
	for (int n = 0; n < N; n++) {
		int c,d;
		cin >> c >> d;
		B.push_back(ii(c, d));
	}
	for (int x = 0; x < 2 * N; x++) {
		for (int y = 0; y < 2 * N; y++) {
			smap[x][y] = 0;
		}
	}
}

void amap() {
	for (int y = 0; y < 2 * N; y++) {
		for (int x = 1; x < 2 * N; x++)smap[x][y] += smap[x - 1][y];
	}
	for (int x = 0; x < 2 * N; x++) {
		for (int y = 1; y < 2 * N; y++)smap[x][y] += smap[x][y - 1];
	}
}
