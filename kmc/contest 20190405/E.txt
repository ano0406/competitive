#include <iostream>
using namespace std;

void solve();

int main() {
	solve();
	return 0;
}

//////////////////////////////////////////////////
//////////////////////////////////////////////////

void solve() {
	while (true) {
		int n, x;
		cin >> n >> x;
		if (n == 0 && x == 0)return;
		int ans = 0;
		for (int a = 1; a <= n; a++) {
			for (int b = a + 1; b <= n; b++) {
				int c = x - a - b;
				if (c > b && c <= n)ans++;
			}
		}
		cout << ans << endl;
	}
}
