using namespace std;
using namespace boost::multiprecision;

void solve() {
	int N;
	cin >> N;
	int sum = 0;
	for (int n = 0; n < N; n++) {
		int a, b;
		cin >> a >> b;
		sum = max(sum, a + b);
	}
	cout << sum << endl;
	return;
}
