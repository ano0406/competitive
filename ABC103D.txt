int N, M;
typedef pair<int, int> _pair;
vector<_pair> arr;

void solve() {
	cin >> N >> M;
	arr.resize(M);
	for (int i = 0; i < M; i++) {
		int a, b;
		cin >> a >> b;
		arr[i] = _pair(--b,--a);
	}
	sort(arr.begin(),arr.end());
	int ans = 0;
	int end = 0;
	for (int i = 0; i < M; i++) {
		if (arr[i].second >= end) {
			ans++;
			end = arr[i].first;
		}
	}
	cout << ans;
}
