string S, T;

bool operator<(string a, string b) {
	for (int i = 0; i < min(a.size(), b.size()); i++) {
		if (a[i] != b[i])return (a[i] < b[i]);
	}
	return (a.size() < b.size());
}


void solve() {
	cin >> S >> T;
	string ans("zzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzzz");
	bool ansflag = false;
	for (int s = 0; s <= S.size() - T.size(); s++) {
		//s•¶Žš–Ú‚ÉT‚ª“ü‚é‚©Šm‚©‚ß‚é
		bool flag = true;
		for (int t = 0; t < T.size(); t++) {
			if (S[s + t] != '?' && S[s + t] != T[t]) {
				flag = false;
				break;
			}
		}
		if (!flag)continue;
		string temp("");
		for (int i = 0; i < s; i++) {
			if (S[i] == '?')temp += string("a");
			else temp += S.substr(i, 1);
		}
		temp += T;
		for (int i = s + T.size(); i < S.size(); i++) {
			if (S[i] == '?')temp += string("a");
			else temp += S.substr(i, 1);
		}
		if (temp < ans)ans = temp;
		ansflag = true;
	}
	if (!ansflag)cout << "UNRESTORABLE";
	else cout << ans;
}
