vector<int> P;
int N;

void solve() {
	cin >> N;
	P.resize(N + 1);
	for (int n = 1; n <= N; n++)cin >> P[n];
	int count = 0;
	for (int n = 1; n <= N; n++) {
		if (P[n] != n)count++;
	}
	if (count == 0 || count == 2)cout << "YES" << endl;
	else cout << "NO" << endl;
	return;
}
