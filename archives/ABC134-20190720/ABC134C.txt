#include <iostream>
#include <vector>
#include <algorithm>
#include <functional>
#include <limits.h>
#include <math.h>
#include <iomanip>
using namespace std;
typedef pair<int, int> pii;
typedef long double ld;
typedef long long ll;

vector<int> A;
vector<int> MAXF, MAXB;

void solve() {
	int N;
	cin >> N;
	A.resize(N + 1);
	for (int n = 1; n <= N; n++)cin >> A[n];
	MAXF.resize(N + 2);
	MAXB.resize(N + 2);
	MAXF[1] = A[1];
	for (int n = 2; n <= N; n++)MAXF[n] = max(A[n], MAXF[n - 1]);
	MAXB[N] = A[N];
	for (int n = N - 1; n >= 1; n--)MAXB[n] = max(A[n], MAXB[n + 1]);
	MAXF[0] = 0; MAXF[N + 1] = 0;
	MAXB[0] = 0; MAXB[N + 1] = 0;
	for (int n = 1; n <= N; n++)cout << max(MAXF[n - 1], MAXB[n + 1]) << endl;
	return;
}
