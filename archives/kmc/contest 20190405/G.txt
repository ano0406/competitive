#include <iostream>
#include <math.h>
#include <algorithm>
using namespace std;

void solve();

int main() {
	solve();
	return 0;
}

//////////////////////////////////////////////////
//////////////////////////////////////////////////

int N, K;
int x[1000];
int y[1000];

void solve() {
	cin >> N >> K;
	for (int k = 0; k < K; k++)cin >> x[k] >> y[k];
	for (int k = 0; k < K; k++) {
		if (x[k] >= floor((N - 1) / 2))x[k] = N + 1 - x[k];
		if (y[k] >= floor((N - 1) / 2))y[k] = N + 1 - y[k];
		int r = min(x[k], y[k]) % 3;
		if (r != 0)cout << r << endl;
		else cout << 3 << endl;
	}
}
