using namespace std;

string S;

void solve() {
	cin >> S;
	int counter = 0;
	for (int s = 0; s < S.size(); s++) {
		if (S[s] == 'x')counter++;
	}
	if (counter >= 8)cout << "NO" << endl;
	else cout << "YES" << endl;
	return;
}
