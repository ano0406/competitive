#include <iostream>
#include <algorithm>
#include <vector>
#include <set>
#include <queue>
#include <functional>
#include <map>
#include <string>

using namespace std;

typedef pair<int, int> pii;
int dp[100000];
int Q,N;
vector<int> ans;

void solve() {
	string s;
	cin >> N >> Q >> s;
	dp[0] = 0;
	for (int i = 1; i < N; i++) {
		if (s[i] == 'C' && s[i - 1] == 'A')dp[i] = dp[i - 1] + 1;
		else dp[i] = dp[i - 1];
	}
	for (int i = 0; i < Q; i++) {
		int l, r;
		cin >> l >> r;
		ans.push_back(dp[r-1] - dp[l-1]);
	}
	for (auto itr = ans.begin(); itr != ans.end(); itr++) {
		cout << *itr << endl;
	}
	return;
}
