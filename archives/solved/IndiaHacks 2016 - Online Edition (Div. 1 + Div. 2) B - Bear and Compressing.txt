int MOD = 1e9 + 7;

long long N, Q;
vector<string> A;
vector<string> B;

long long calc(string s) {
	if (s.size() == N)return 1;
	long long ans = 0;
	for (long long q = 0; q < Q; q++) {
		if (s[0] == B[q][0]) {
			ans += calc(A[q] + s.substr(1));
		}
	}
	return ans;
}

int main() {
	cin >> N >> Q;
	A.resize(Q);
	B.resize(Q);
	for (long long q = 0; q < Q; q++) {
		string s;
		cin >> s;
		if (s[0] < 'a' || s[0] > 'f')continue;
		if (s[1] < 'a' || s[1] > 'f')continue;
		string s2;
		cin >> s2;
		if (s2[0] < 'a' || s2[0] > 'f')continue;
		A.push_back(s);
		B.push_back(s2);
	}
	Q = A.size();
	long long ans = calc(string("a"));
	cout << ans << endl;
	return 0;
}
