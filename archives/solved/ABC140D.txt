void solve() {
	int N, K;
	string S;
	cin >> N >> K >> S;
	int counter = 0;
	for (int n = 1; n < N; n++) {
		if (S[n] != S[n - 1])counter++;
	}
	counter++;
	while (K > 0 && counter > 1) {
		counter -= 2;
		if (counter == 0)counter = 1;
		K--;
	}
	cout << N - counter << endl;
	return;
}

		solve();
		cout << "####################" << endl;
	}
#else
	solve();
#endif
	return 0;
}
