template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

int main() {
	int N, K, Q;
	vector<int> A;
	vector<int> dp;
	cin >> N >> K >> Q;
	A.resize(Q);
	in(A);
	dp.resize(N,K - Q);
	for (auto itr = A.begin(); itr != A.end();itr++) {
		dp[--(*itr)]++;
	}
	for (int n = 0; n < N; n++) {
		if (dp[n] > 0)cout << "Yes" << endl;
		else cout << "No" << endl;
	}
	return 0;
}
