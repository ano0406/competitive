using namespace std;
 
long long S, X;
 
int main() {
	cin >> S >> X;
	int d = 0;
	if (S == X)d = -2;
	long long ans = 1;
	for (int c = 0; c < 60; c++) {
		if ((S & 1) != (X & 1)) {
			cout << 0 << endl;
			return 0;
		}
		else if ((X & 1) == 1) {
			ans *= 2;
			X >>= 1;
			S >>= 1;
		}
		else {
			if ((X & 2) != (S & 2)) {
				X >>= 1;
				S >>= 1;
				S--;
			}
			else {
				X >>= 1;
				S >>= 1;
			}
		}
	}
	cout << ans + d << endl;
	return 0;
}
