int MOD = 1e9 + 7;
 
long long modpow(long long base, long long p) {
	if (p == 0)return 1;
	else if (p == 1)return base;
	long long base2 = modpow(base, p / 2);
	base2 = (base2 * base2) % MOD;
	if (p & 1)base2 = (base2 * base) % MOD;
	return base2;
}
 
int main() {
	long long A, B;
	cin >> A >> B;
	long long first = modpow(2, B);
	first = (first + MOD - 1) % MOD;
	cout << modpow(first, A) << endl;
	return 0;
}
