int main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	ll N;
	cin >> N;
	ll past;
	cin >> past;
	for (ll n = 1; n < N; n++) {
		ll cur;
		cin >> cur;
		if (past == cur)cout << "stay\n";
		else if (past > cur) {
			cout << "down " << past - cur << "\n";
		}
		else cout << "up " << cur - past << "\n";
		past = cur;
	}
	return 0;
}
