long long MOD = 1e9 + 7;
typedef long long ll;
typedef pair<ll, ll> pll;

ll N, M;
vector<ll> fav;

int main() {
	cin >> N >> M;
	fav.resize(M);
	for (ll m = 0; m < N; m++) {
		ll K;
		cin >> K;
		for (ll k = 0; k < K; k++) {
			ll a;
			cin >> a;
			fav[--a]++;
		}
	}
	ll count = 0;
	for (ll n = 0; n < M; n++) {
		if (fav[n] == N)count++;
	}
	cout << count << endl;
	return 0;
}
