int N;
vector<int> A, B,atemp;

void solve() {
	cin >> N;
	A.resize(1e5 + 1, 0);
	B.resize(1e5 + 1, 0);
	atemp.resize(N + 1);
	for (int n = 1; n <= N; n++) {
		cin >> atemp[n];
	}
	for (int n = 1; n <= N; n++) {
		int b;
		cin >> b;
		B[b]++;
	}
	int prevans = INT_MAX;
	for (int n = 1; n <= N; n++) {
		A[atemp[n]]++;
		prevans = min(prevans, B[atemp[n]] / A[atemp[n]]);
		cout << prevans << endl;
	}
}
