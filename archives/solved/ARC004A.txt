#include <iostream>
#include <math.h>

int N;
int x[128], y[128];
using namespace std;

int main() {
	cin >> N;
	for (int i = 0; i < N; i++) cin >> x[i] >> y[i];
	float l = 0;
	for (int i = 0; i < N; i++) {
		for (int j = i + 1; j < N; j++) {
			float ltemp = sqrtf((float)((x[i] - x[j])*(x[i] - x[j]) + (y[i] - y[j])*(y[i] - y[j])));
			if (l < ltemp) {
				l = ltemp;
			}
		}
	}
	cout << l << endl;
	return 0;
}
