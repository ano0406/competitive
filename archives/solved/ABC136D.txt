using namespace std;
long long MOD = 1e9 + 7;

int N;
string S;
int L[100000];
int R[100000];
int ans[100000];

int main() {
	cin >> S;
	N = S.size();
	for (int n = 0; n < N; n++) {
		if (S[n] == 'R')R[n] = n;
		else L[n] = n;
	}
	for (int n = 1; n < N; n++) {
		R[n] = max(R[n], R[n - 1]);
	}
	for (int n = N - 1; n >= 0; n--) {
		if (L[n] == 0)L[n] = L[n + 1];
	}
	for (int n = 0; n < N; n++)ans[n] = 0;
	for (int n = 0; n < N; n++) {
		if (S[n] == 'R') {
			int t = L[n] - n;
			if (t % 2 == 0)ans[L[n]]++;
			else ans[L[n] - 1]++;
		}
		else {
			int t = n - R[n];
			if (t % 2 == 0)ans[R[n]]++;
			else ans[R[n] + 1]++;
		}
	}
	for (int n = 0; n < N; n++) {
		cout << ans[n] << " ";
	}
	cout << endl;
	return 0;
}
