void solve() {
	int N, X;
	cin >> N >> X;
	vector<int> A;
	A.resize(N);
	int asum = 0;
	int counter = 0;
	for (int n = 0; n < N; n++) {
		int a;
		cin >> a;
		if (a != -1)asum ^= a;
		else counter++;
		A[n] = a;
	}
	if (counter == 0) {
		if (asum != X)cout << -1 << endl;
		else {
			for (int n = 0; n < N; n++)cout << A[n] << " ";
			cout << endl;
		}
		return;
	}
	if (counter == 1) {
		if ((asum ^ X) > X)cout << -1 << endl;
		else {
			for (int n = 0; n < N; n++) {
				if (A[n] != -1)cout << A[n] << " ";
				else cout << (X ^ asum) << " ";
			}
			cout << endl;
		}
		return;
	}
	counter = 0;
	int y1 = (asum ^ X) & X;
	int y2 = (asum ^ X) ^ y1;
	if (y2 > X)cout << -1 << endl;
	else {
		for (int n = 0; n < N; n++) {
			if (A[n] != -1)cout << A[n] << " ";
			else {
				if (counter == 0)cout << y1 << " ";
				else if (counter == 1)cout << y2 << " ";
				else cout << 0 << " ";
				counter++;
			}
		}
		cout << endl;
	}
	return;
}

	return 0;
}
