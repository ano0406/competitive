using namespace std;

int N, T;
vector<int> A;

int main() {
	cin >> N >> T;
	A.resize(N);
	for (int n = 0; n < N; n++)cin >> A[n];
	sort(A.begin(), A.end(), [](int a, int b) {
		return a > b;
	});
	int counter = 0;
	for (int n = 0; n < N; n++) {
		if (A[n] + T >= A[0])counter++;
	}
	cout << counter << endl;
	return 0;
}
