string S;

int main() {
	cin >> S >> K;
	while (K >= 0) {
		if (K == 1) {
			cout << S.substr(0, 1) << endl;
			return 0;
		}
		else if (S[0] == '1') {
			K--;
			S = S.substr(1);
		}
		else {
			cout << S[0] << endl;
			return 0;
		}
	}
	return 0;
}
