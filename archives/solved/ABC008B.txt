	for (int n = 0; n < N; n++) {
		string name;
		cin >> name;
		if (M.find(name) == M.end())M.insert(pair < string, long long>(name, 1));
		else M.find(name)->second++;
	}
	string maxname;

	for (auto e : M) {
		if (e.second > maxnum) {
			maxnum = e.second;

		}
	}
