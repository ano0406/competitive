int M, D;

void solve() {
	cin >> M >> D;
	int counter = 0;
	for (int d = 10; d <= D; d++) {
		int d1 = d % 10;
		int d10 = (d - d1) / 10;
		if (d1 < 2 || d10 < 2)continue;
		if (d1 * d10 <= M)counter++;
	}
	cout << counter << endl;
	return;
}
