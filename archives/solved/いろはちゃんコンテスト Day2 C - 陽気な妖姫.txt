const long long MOD = 1e9 + 7;
typedef long long ll;
typedef long double ld;
typedef pair<ll, ll> pll;
typedef pair<ld, ll> pdl;
typedef pair<ld, ld> pdd;
typedef vector<ll> VLL;
typedef vector<VLL> VVLL;
//typedef boost::multiprecision::cpp_int bigint;

int main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	ll N;
	cin >> N;
	vector<pll> V0(N);
	for (ll n = 0; n < N; n++) {
		cin >> V0[n].first;
		V0[n].second = n;
	}
	VLL ans(N);
	sort(V0.begin(), V0.end(), [](pll a, pll b) {
		return a.first < b.first;
	});
	ans[V0[0].second] = 1;
	for (ll n = 1; n < N; n++) {
		if (V0[n].first == V0[n-1].first) {
			ans[V0[n].second] = ans[V0[n - 1].second];
		}
		else {
			ans[V0[n].second] = ans[V0[n - 1].second] + 1;
		}
	}
	for (ll n = 0; n < N; n++) {
		cout << ans[n] << "\n";
	}
	return 0;
}
