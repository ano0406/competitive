bool judge(ll n) {
	return (n % 2 != 0 && n % 3 != 2);
}

int main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	ll N;
	cin >> N;
	VLL A(N);
	for (ll n = 0; n < N; n++)cin >> A[n];
	ll ans = 0;
	for (ll n = 0; n < N; n++) {
		while (A[n]>=1&&!judge(A[n])) {
			ans++;
			A[n]--;
		}
	}
	cout << ans << "\n";
