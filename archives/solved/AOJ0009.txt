//素数判定
//O(√n)
bool isPrime(int n) {
	for (int i = 2; i <= floor(sqrt(n)); i++) {
		if (n%i == 0)return false;
	}
	return true;
}

//n未満の自然数のでのエラトステネスの篩
//arr[n]=true -> 素数
//O(nlog(logn)) , 実質O(n)
bool* eratosthenes(int n) {
	bool* array = new bool[n];
	for (int i = 0; i < n; i++)array[i] = true;
	for (int i = 2; i < n; i++) {
		if (!array[i])continue;
		int t = 2*i;
		while (t < n) {
			array[t] = false;
			t += i;
		}
	}
	array[0] = false;
	array[1] = false;
	return array;
}

int isp[1000000];
vector<int> v;

void solve() {
	bool* arr = eratosthenes(1000000);
	isp[0] = 0;
	for (int n = 1; n < 1000000; n++) {
		isp[n] = isp[n - 1] + (arr[n]?1:0);
	}
	while (!cin.eof()) {
		int n;
		cin >> n;
		if (cin.fail())break;
		v.push_back(isp[n]);
	}
	for (auto itr = v.begin(); itr != v.end(); itr++) cout << *itr << endl;
	return;
}
