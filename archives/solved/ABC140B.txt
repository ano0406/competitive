int N;

void solve() {
	vector<int> A, B, C;
	int ans = 0;
	cin >> N;
	A.resize(N);
	for (int n = 0; n < N; n++) {
		cin >> A[n];
		A[n]--;
	}
	B.resize(N);
	for (int n = 0; n < N; n++) {
		cin >> B[n];
		ans += B[n];
	}
	C.resize(N - 1);
	for (int n = 0; n < N - 1; n++)cin >> C[n];
	for (int n = 0; n < N - 1; n++) {
		if (A[n + 1] == A[n] + 1)ans += C[A[n]];
	}
	cout << ans << endl;
	return;
}

		solve();
		cout << "####################" << endl;
	}
#else
	solve();
#endif
	return 0;
}
