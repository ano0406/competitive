ll N;
vector<ll> A;

int main() {
	cin >> N;
	A.resize(N);
	ll asum = 0;
	for (ll n = 0; n < N; n++) {
		ll a, b;
		cin >> a >> b;
		asum += a;
		A[n] = a + b;
	}
	sort(A.begin(), A.end(), [](ll a, ll b) {return a > b; });
	for (ll n = 1; n < N; n += 2) {
		asum -= A[n];
	}
	cout << asum << endl;
	return 0;
}
