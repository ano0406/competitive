template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

long long N, K, X, Y;

int main() {
	cin >> N >> K >> X >> Y;
	if (N <= K)cout << X * N << endl;
	else cout << X * K + (N-K) * Y << endl;
	return 0;
}
