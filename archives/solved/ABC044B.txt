template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

string W;

int main() {
	cin >> W;
	bool flag = true;
	for (int c = 'a'; c <= 'z'; c++) {
		int counter = 0;
		for (int n = 0; n < W.size(); n++) {
			if (W[n] == c)counter++;
		}
		if (counter % 2 != 0) {
			flag = false;
			break;
		}
	}
	if (flag)cout << "Yes" << endl;
	else cout << "No" << endl;
	return 0;
}
