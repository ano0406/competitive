#include <iostream>
#include <math.h>
#include <algorithm>
#include <vector>
using namespace std;

void solve();

int main() {
	solve();
	return 0;
}

//////////////////////////////////////////////////
//////////////////////////////////////////////////

int T, N, M;
vector<int> A, B;

void solve() {
	cin >> T >> N;
	A.resize(N);
	for (int n = 0; n < N; n++)cin >> A[n];
	cin >> M;
	B.resize(M);
	for (int m = 0; m < M; m++)cin >> B[m];
	while (!B.empty()) {
		bool flag = false;
		for (int n = 0; n < A.size(); n++) {
			if (B[0] - T <= A[n] && B[0] >= A[n]) {
				B.erase(B.begin(),B.begin()+1);
				A.erase(A.begin()+n,A.begin()+n+1);
				flag = true;
				break;
			}
		}
		if (!flag) {
			cout << "no" << endl;
			return;
		}
	}
	cout << "yes" << endl;
	return;
}
