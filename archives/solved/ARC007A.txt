template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

string missing;
string input;

int main() {
	cin >> missing >> input;
	string ans("");
	for (int i = 0; i < input.size(); i++) {
		if (input.substr(i, 1) != missing)ans += input.substr(i, 1);
	}
	cout << ans << endl;
	return 0;
}
