ll N;
vector<ll> A;

int main() {
	cin >> N;
	A.resize(N);
	for (ll n = 0; n < N; n++)cin >> A[n];
	ll counter = 1;
	ll ans = 0;
	for (ll n = 1; n < N; n++) {
		if (A[n] <= A[n - 1]) {
			ans += counter * (counter + 1) / 2;
			counter = 1;
		}
		else counter++;
	}
	ans += counter * (counter + 1) / 2;
	cout << ans << endl;

}
