int N;
vector<int> A;
void solve() {
	cin >> N;
	A.clear();
	A.resize(N);
	for (int n = 0; n < N; n++)cin >> A[n];
	bool flag = false;
	for (int n = 0; n < N; n++) {
		if (A[n] % 2 == 1) {
			flag = true;
			break;
		}
	}
	if (flag)cout << "first" << endl;
	else cout << "second" << endl;
	return;
}

int main() {
#ifdef DEBUGTIME
	for (int time = 0; time < DEBUGTIME; time++) {
		solve();
		cout << "####################" << endl;
	}
#else
	solve();
#endif
	return 0;
}
