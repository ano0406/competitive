long long MOD = 1e9 + 7;

int N, M;
int imos[101][101][101];

int main() {
	cin >> N >> M;
	for (int x = 0; x <= 100; x++) {
		for (int y = 0;y <= 100; y++) {
			for (int z = 0; z <= 100; z++)imos[x][y][z] = 0;
		}
	}
	for (int n = 0; n < N; n++) {
		int a, b, c, w;
		cin >> a >> b >> c >> w;
		imos[a][b][c] = max(imos[a][b][c], w);
	}
	for (int a = 0; a <= 100; a++) {
		for (int b = 0; b <= 100; b++) {
			for (int c = 0; c <= 100; c++) {
				if (a > 0)imos[a][b][c] = max(imos[a][b][c], imos[a - 1][b][c]);
				if (b > 0)imos[a][b][c] = max(imos[a][b][c], imos[a][b - 1][c]);
				if (c > 0)imos[a][b][c] = max(imos[a][b][c], imos[a][b][c - 1]);
			}
		}
	}
	for (int m = 0; m < M; m++) {
		int a, b, c;
		cin >> a >> b >> c;
		cout << imos[a][b][c] << endl;
	}
	return 0;
}
