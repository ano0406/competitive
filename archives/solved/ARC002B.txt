template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

int dates[2][13] = { { 1,31,28,31,30,31,30,31,31,30,31,30,31 } 
,{ 1,31,29,31,30,31,30,31,31,30,31,30,31 } };

int main() {
	string s0;
	cin >> s0;
	int Y, M, D;
	Y = stoi(s0.substr(0, 4));
	M = stoi(s0.substr(5, 2));
	D = stoi(s0.substr(8, 2));
	while (Y % (M * D) != 0) {
		int i = 0;
		if (Y % 4 == 0)i = 1;
		if (Y % 100 == 0)i = 0;
		if (Y % 400 == 0)i = 1;
		D++;
		if (D > dates[i][M]) {
			D = 1;
			M++;
		}
		if (M == 13) {
			M = 1;
			Y++;
		}
	}
	cout << Y << "/";
	if (M >= 10)cout << M << "/";
	else cout << 0 << M << "/";
	if (D >= 10)cout << D << endl;
	else cout << 0 << D << endl;
	return 0;
}
