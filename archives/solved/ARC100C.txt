using namespace std;

int N;
vector<int> A;

int main() {
	cin >> N;
	A.resize(N+1);
	A[0] = INT_MIN;
	for (int n = 1; n <= N; n++) {
		cin >> A[n];
		A[n] -= n;
	}
	sort(A.begin(), A.end());
	int b;
	if (N % 2 == 0)b = A[N / 2];
	else b = A[N / 2 + 1];
	long long ans = 0;
	for (int n = 1; n <= N; n++) {
		if (A[n] < b)ans += (long long)b - A[n];
		else ans += (long long)A[n] - b;
	}
	cout << ans << endl;
	return 0;
}
