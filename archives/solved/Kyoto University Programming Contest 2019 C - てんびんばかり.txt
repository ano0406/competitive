int M, K;
vector<long long> V;

int main() {
	cin >> M >> K;
	long long sum = 0;
	int N = 0;
	for (int k = 0; k < K; k++) {
		V.push_back(1);
		sum += 1;
	}
	N = 1;
	while (sum < M) {
		for (int k = 0; k < K; k++)V.push_back(2 * sum + 1);
		sum += K * V.back();
		N++;
	}
	cout << N << endl;
	return 0;
}
