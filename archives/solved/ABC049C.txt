#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

void solve() {
	string S;
	cin >> S;
	while (S.size() > 0) {
		if (S.size() >= 7) {
			if (S.substr(S.size() - 7, 7) == string("dreamer")) {
				S.erase(S.size() - 7, 7);
				continue;
			}
		}
		if (S.size() >= 5) {
			if (S.substr(S.size() - 5, 5) == string("dream")) {
				S.erase(S.size() - 5, 5);
				continue;
			}
		}
		if (S.size() >= 6) {
			if (S.substr(S.size() - 6, 6) == string("eraser")) {
				S.erase(S.size() - 6, 6);
				continue;
			}
		}
		if (S.size() >= 5) {
			if (S.substr(S.size() - 5, 5) == string("erase")) {
				S.erase(S.size() - 5, 5);
				continue;
			}
		}
		break;
	}
	if (S == string(""))cout << "YES" << endl;
	else cout << "NO" << endl;
	return;
}
