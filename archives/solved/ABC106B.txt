long long MOD = 1e9 + 7;

int main() {
	int N;
	cin >> N;
	int ans = 0;
	for (int n = 1; n <= N; n += 2) {
		int counter = 0;
		for (int p = 1; p * p <= n; p++) {
			if (n % p == 0) {
				if (p * p == n)counter++;
				else counter += 2;
			}
		}
		if (counter == 8)ans++;
	}
	cout << ans << endl;
	return 0;
}
