void solve();

int main() {
	solve();
	return 0;
}

#include <iostream>
#include <vector>
#include <limits.h>
#include <algorithm>
#include <string>
#include <math.h>
#include <limits.h>

using namespace std;

int N;
vector<long double> A;
void solve() {
	cin >> N;
	A.resize(N);
	for (int n = 0; n < N; n++)cin >> A[n];
	long double s = 0, e = 1e14;
	while (e - s > 1e-6) {
		long double x0 = (e + s) / 2;
		long double x = x0;
		for (int n = 0; n < N - 1; n++)x = A[n] - x;
		if (A[N - 1] < x + x0)e = x0;
		else s = x0;
	}
	vector<int> x;
	x.resize(N);
	x[0] = (int)round(s);
	for (int n = 1; n < N; n++)x[n] = A[n - 1] - x[n - 1];
	for (int n = 0; n < N; n++) {
		cout << 2*x[n] << endl;
	}
	return;
}
