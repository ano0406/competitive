#include <iostream>
#include <math.h>
#include <algorithm>
#include <vector>
using namespace std;

void solve();

int main() {
	solve();
	return 0;
}

//////////////////////////////////////////////////
//////////////////////////////////////////////////
int N;
int P[100];
bool dp[100][10001];

void solve() {
	cin >> N;
	for(int n = 0;n < N;n++){
		cin >> P[n];
	}
	for (int i = 0; i <= 10000; i++)dp[0][i] = false;
	dp[0][0] = true;
	dp[0][P[0]] = true;
	for (int n = 1; n < N; n++) {
		for (int i = 0; i <= 10000; i++) {
			if (dp[n - 1][i]) {
				dp[n][i] = true;
				if(i + P[n] <= 10000)dp[n][i + P[n]] = true;
			}
		}
	}
	int ans = 0;
	for (int i = 0; i <= 10000; i++)if (dp[N - 1][i])ans++;
	cout << ans << endl;
	return;
}
