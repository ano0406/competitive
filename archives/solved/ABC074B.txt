int N, K;
vector<int> X;

int main() {
	cin >> N >> K;
	X.resize(N);
	for (int n = 0; n < N; n++)cin >> X[n];
	int ans = 0;
	for (int n = 0; n < N; n++) {
		ans += min(X[n], K - X[n]);
	}
	cout << 2 * ans << endl;
	return 0;
}
